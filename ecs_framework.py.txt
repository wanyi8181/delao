import esper
import pygame

# 初始化 Pygame
pygame.init()
screen = pygame.display.set_mode((800, 600))
pygame.display.set_caption("2D地牢游戏-ECS架构版")

# 定义 ECS 组件
class Position(esper.Component):
    def __init__(self, x: int = 0, y: int = 0):
        self.x = x
        self.y = y

class Velocity(esper.Component):
    def __init__(self, dx: int = 0, dy: int = 0):
        self.dx = dx
        self.dy = dy

class Renderable(esper.Component):
    def __init__(self, surface: pygame.Surface):
        self.surface = surface

# 定义处理器（核心逻辑）
class MovementProcessor(esper.Processor):
    def process(self, dt: float = 1.0):
        for ent, (pos, vel) in self.world.get_components(Position, Velocity):
            pos.x += vel.dx * dt
            pos.y += vel.dy * dt

class RenderProcessor(esper.Processor):
    def __init__(self, screen: pygame.Surface):
        self.screen = screen

    def process(self, dt: float = 1.0):
        self.screen.fill((0, 0, 0))
        for ent, (pos, rend) in self.world.get_components(Position, Renderable):
            self.screen.blit(rend.surface, (pos.x, pos.y))
        pygame.display.flip()

# 初始化 ECS 世界与处理器
world = esper.World()
world.add_processor(MovementProcessor())
world.add_processor(RenderProcessor(screen), priority=1)

# 创建游戏实体（玩家）
player_surf = pygame.Surface((32, 32))
player_surf.fill((0, 255, 0))
player = world.create_entity()
world.add_component(player, Position(400, 300))
world.add_component(player, Velocity(2, 0))
world.add_component(player, Renderable(player_surf))

# 游戏主循环
running = True
clock = pygame.time.Clock()
while running:
    for event in pygame.event.get():
        if event.type == pygame.QUIT:
            running = False

    # 执行 ECS 处理器逻辑
    world.process(1.0)
    clock.tick(60)

pygame.quit()
